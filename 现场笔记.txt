	var React = require('react-native');
	var { Text } = React;
		*指定了三个对象
	var GeoInfo = React.createClass({
		getInitialState: function() {
		return { position: 'unknown' };
	},	
	
	componentDidMount: function() {
		navigator.geolocation.getCurrentPosition(
			(position) => this.setState({position}),
			(error) => console.error(error)
		);
	},
	
	render: function() {
	   return (
		  <Text>
			Position: {JSON.stringify(this.state.position)}
		  </Text>
		);
	   },
	});

	#CSS语法#
		标准流:一种标准的网页布局样式,是一种模板
		浮动:属性,用于页面布局的重要手段
		定位:属性,用于页面布局的重要手段
		
		
		DOCTYPE:指定html文本的解析方式	

		
		meta: 提供关于页面的"元信息"的标签 位于head内,主要属性有
		
				content 定义与http-equiv相关的元信息,和http-equiv配对使用,表示值.具体解释见http-equiv词条
				
				http-equiv 表示键值对里的键  发文档给浏览器时,在内容的头部一定要加上,从而告诉浏览器文档的类型
						<meta http-equiv="charset" content="iso-8859-1">
						<meta http-equiv="expires" content="31 Dec 2008">
						则在发给浏览器的文档的头部,会包含
						charset:iso-8859-1
						expires:31 Dec 2008

						
						
				name 	是键值对里的键名称,作用同http-equiv 例如 <meta name="keywords"  content="HTML,ASP,PHP,SQL" > 
						服务器传文档给浏览器时,头部就会包含
						keywords:HTML,ASP,PHP,SQL 的信息
						
				scheme  定义用于翻译content属性的格式
		
				
		倘若出现中文乱码,将<meta>标签里的charset属性设为 utf-8一般能解决		
		
				body:
					是页面上呈现的元素的父节点
				
					
			标准文件流
					在不用CSS时,各元素的排列样式.就是默认样式
					元素分为
						块元素
							同级的元素都是竖直方向左对齐的.典型样例:<ul><li></li></ul>
						
						内联元素
							类似普通文本样式,元素横向排列,到最右时折行
						
							
			在用DOM思想阅读时,应当由外而内地分析
					以表结构为例
					    <ul>
       					     <li>1st list 1st item</li>
        					 <li class="withborder">
            					    1st list 2st item,wrap required so content is longer
        					 </li>
    					 </ul>
    					 <ul>
       					     <li>2st list 1st item</li>
      					      <li class="withborder">
       					         2st list 2st item,wrap required so content is longer
      					      </li>
  					      </ul>	
  					      
					第一步
						body是最大的块元素,有一个
						body下一级有两个ul,是块元素,所以垂直方向左对齐地排列
					
  					第二步
  					 	ul也是个容器,在容器里有两个<li>,是块元素,所以竖直方向左对齐地排列
  					 	
  					 第三步
  					 	如果有样式,那就拿样式去修饰对应的元素
  					 	
  			
  			重要的块元素 div
  					是一种块容器标识,里面可以容纳各种各样的元素.常见用法是将div里的内容视为独立对象,用于css的控制.
  					
  			重要的内联元素  span
  					可以像div一样用,用于控制span围住的标识的样式.但span无法指定高度宽度,只能指定样式属性
  					
  			height属性:
  					尽量避免使用,因为在IE和firefox中会出现表现不同的问题,一般用min-height, min-width, max-width等来限定尺寸
  					
  			
  			margin属性:
  					用于指定盒子的间距.并且可以为负数,为负数时,元素会出现部分交叠

  					
  			float属性:
  					用于更改标准流里指定的尺寸,相对位置等要素.
  					如设其为left,则块会左对齐,并且不设right-margin,而是根据content大小来决定大小
  					
  					例如,在head的style里
  					    .son1{
          					    /* 写浮动样式 */
          					    float:left;
      					}
      					.son2{
      					  	标准流样式
     					}
     					
     					在body里
  					       <div class="son1">Box-1</div>
  					       <div class="son2">Box-2</div>
  					 	
  					     则此时类为son1的标签已脱离标准流,被标准流无视
  					     son2标签会顶到容器的最上面
  					     而son1则根据浮动的规则,顶到了最上面,这时box2
  					     并且son1由于是浮动,border不再撑开,而是围绕在内容四周
  					     
  					     float四大属性
  					     	1 包裹性 如果没设置高度和宽度,会尽量缩小
  					     	2 浮动破坏性 它的下一个标准流元素里的非block元素会环绕
  					     	3 跟随性  浮动元素的下一个元素也浮动 则下一个元素会跟随前一个显示成一行
  					     	
  					     	
  					     
  					 	float可改变标准流中由出现顺序实现的排序
  					 	例如
  					 		.son1{
  					 			float : right;
  					 		}
  					 	
  					 		.son2{
  					 			float : left;
  					 		}

  					 		<div class="son1">A</div>
  					 		<div class="son2">B</div>
  					     	
  					 		则son2 紧贴容器左侧
  					 		则son1 紧贴容器右侧
  					 		
  					 		在UI设计中,可利用此特性重布页面,将重要的内容float:left 以优化用户体验
  					 		
  					 		不过在页面变窄,需要浮动元素折行时,却是按排列顺序来折行的.排在前面的元素先折行
  					 		
  					 		
  					 	position属性为relative时:
  					 		对标准流里的元素,会使其相对默认位置位移.而不影响其他元素.它仍是标准流元素.这点和float大有区别,切记切记.
  					 		
  					 	
    					position属性为absolute时:
    						会找到在层级上离它最近的为absolute/relative/元素并定位,
